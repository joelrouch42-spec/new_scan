//
// Combined Squeeze Momentum + MACD Signal Indicator with ADX Filter
// Affiche une flèche VERTE au moment du changement vers vert+vert (en tendance)
// Affiche une flèche ROUGE au moment du changement vers rouge+rouge (en tendance)
//
//@version=5
indicator(shorttitle="SQZMOM+MACD+ADX", title="Combined Squeeze & MACD Signals with ADX Filter", overlay=true)

// ========== SQUEEZE MOMENTUM ==========
length = input(20, title="BB Length")
mult = input(2.0, title="BB MultFactor")
lengthKC = input(20, title="KC Length")
multKC = input(1.5, title="KC MultFactor")
useTrueRange = input(true, title="Use TrueRange (KC)")

// Calculate BB
source = close
basis = ta.sma(source, length)
dev = multKC * ta.stdev(source, length)
upperBB = basis + dev
lowerBB = basis - dev

// Calculate KC
ma = ta.sma(source, lengthKC)
range_val = useTrueRange ? ta.tr : (high - low)
rangema = ta.sma(range_val, lengthKC)
upperKC = ma + rangema * multKC
lowerKC = ma - rangema * multKC

sqzOn = (lowerBB > lowerKC) and (upperBB < upperKC)
sqzOff = (lowerBB < lowerKC) and (upperBB > upperKC)
noSqz = not sqzOn and not sqzOff

val = ta.linreg(source - math.avg(math.avg(ta.highest(high, lengthKC), ta.lowest(low, lengthKC)), ta.sma(close, lengthKC)), lengthKC, 0)

// DEBUG: Afficher la valeur du Squeeze Momentum
plot(val, "Squeeze Momentum Value", color=color.purple, linewidth=2, display=display.data_window)

// Squeeze Momentum: lime (vert vif) ou red (rouge vif)
sqz_isGreen = val > 0 and val > nz(val[1])
sqz_isRed = val < 0 and val < nz(val[1])

// ========== MACD ==========
fastLength = input(12, title="MACD Fast")
slowLength = input(26, title="MACD Slow")
signalLength = input(9, title="MACD Signal")

fastMA = ta.ema(source, fastLength)
slowMA = ta.ema(source, slowLength)
macd = fastMA - slowMA
signal = ta.sma(macd, signalLength)

// MACD: green when above signal, red when below
macd_isGreen = macd >= signal
macd_isRed = macd < signal

// ========== FILTRE ADX ==========
useADX = input(true, title="Use ADX Filter")
adxLength = input(14, title="ADX Length")
adxThreshold = input(25, title="ADX Threshold (min for trend)")

[diPlus, diMinus, adx] = ta.dmi(adxLength, adxLength)
inTrend = adx > adxThreshold

// ========== COMBINED SIGNALS (CHANGEMENTS SEULEMENT) ==========
// État combiné actuel
combined_isGreen = sqz_isGreen and macd_isGreen
combined_isRed = sqz_isRed and macd_isRed

// État combiné précédent
combined_wasGreen = sqz_isGreen[1] and macd_isGreen[1]
combined_wasRed = sqz_isRed[1] and macd_isRed[1]

// Flèche VERTE: transition vers vert+vert ET en tendance
greenSignal = combined_isGreen and not combined_wasGreen and (not useADX or inTrend)

// Flèche ROUGE: transition vers rouge+rouge ET en tendance
redSignal = combined_isRed and not combined_wasRed and (not useADX or inTrend)

// Plot arrows on chart
plotshape(greenSignal, title="Buy Signal", location=location.belowbar, color=color.lime, style=shape.triangleup, size=size.small)
plotshape(redSignal, title="Sell Signal", location=location.abovebar, color=color.red, style=shape.triangledown, size=size.small)

// Plot ADX pour visualiser (optionnel - commenter si vous ne voulez pas le voir)
hline(adxThreshold, "ADX Threshold", color=color.gray, linestyle=hline.style_dashed)
plot(adx, "ADX", color=color.blue, linewidth=1)

// Alerts automatiques en temps réel
if greenSignal
    alert("🟢 ACHAT: Squeeze LIME + MACD GREEN + ADX=" + str.tostring(adx, "#.##"), alert.freq_once_per_bar)

if redSignal
    alert("🔴 VENTE: Squeeze RED + MACD RED + ADX=" + str.tostring(adx, "#.##"), alert.freq_once_per_bar)
